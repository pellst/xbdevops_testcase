CREATE VIEW db2inst1.xt_ss_v AS SELECT et.feature_id AS probe_id , et.experiment_id , es.name AS experiment_source , es.experimentsource_id , et.expressiontest_id , est.expressionstagestested_id, est.specimenstages_id , ss.specimen_id , ss.start_stage_id , ss.end_stage_id , ss.stage_resolution_id , CASE WHEN ( ( sStart.name = 'unspecified stage' ) OR ( sEnd.name = 'unspecified stage' ) ) THEN sStart.stage_number WHEN ( sStart.name = sEnd.name ) THEN sStart.name WHEN ( sStart.name != sEnd.name ) THEN sStart.name || ' to ' || sEnd.stage_number END AS stage_text , sStart.name start_stage_name , sEnd.name end_stage_name , sStart.rank start_stage_rank , sEnd.rank end_stage_rank , sStart.stage_number start_stage_number , sEnd.stage_number end_stage_number , sr.name stage_resolution FROM expressiontest et , experimentsource es , experiment e , expression_stages_tested est, specimen_stages ss , stage sStart , stage sEnd , stage_resolution sr WHERE et.expressiontest_id = est.expressiontest_id AND et.experiment_id = e.experiment_id AND e.experimentsource_id = es.experimentsource_id AND est.specimenstages_id = ss.specimenstages_id AND ss.start_stage_id = sStart.stage_id AND ss.end_stage_id = sEnd.stage_id AND ss.stage_resolution_id = sr.stage_resolution_id
