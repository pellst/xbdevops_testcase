CREATE VIEW DB2INST1.EXPERIMENTSOURCE_incl_del_v (EXPERIMENTSOURCE_ID, NAME, CREATED_AT, MODIFIED_AT, CREATED_BY, MODIFIED_BY,transaction_date, transaction_type) AS SELECT t.EXPERIMENTSOURCE_ID, t.NAME, t.CREATED_AT, t.MODIFIED_AT, t.CREATED_BY, t.MODIFIED_BY, coalesce(t.modified_at, t.created_at) as transaction_time, CASE WHEN t.modified_at IS NULL THEN 'I' ELSE 'U' END AS transaction_date FROM DB2INST1.EXPERIMENTSOURCE t UNION SELECT a.EXPERIMENTSOURCE_ID, a.NAME, a.CREATED_AT, a.MODIFIED_AT, a.CREATED_BY, a.MODIFIED_BY,transaction_date, transaction_type FROM DB2INST1.AUDIT_EXPERIMENTSOURCE a WHERE transaction_type = 'D'
