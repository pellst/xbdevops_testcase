CREATE OR REPLACE VIEW expression_images_a_v as ( select sp.*, ap.expressionassay_id, ap.anatomy_item_id, ap.anatomy_item , CASE WHEN ap.biomedia_id is not null THEN RTRIM(cast(sp.expressionstagestested_id as char(15))) || RTRIM(cast(ap.biomedia_id as char(15))) ELSE RTRIM(cast(sp.expressionstagestested_id as char(15))) END as image_pattern_id from stage_pattern_a_v sp left join anatomy_pattern_a_v ap ON sp.biomedia_id=ap.biomedia_id and sp.expressionstagestested_id=ap.expressionstagestested_id )
