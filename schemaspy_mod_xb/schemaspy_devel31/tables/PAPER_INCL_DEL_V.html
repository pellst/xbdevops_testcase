CREATE VIEW DB2INST1.PAPER_incl_del_v (PAPER_ID, TITLE, PAGINATION, PAPERTYPE_ID, JOURNALISSUE_ID, BOOK_ID, AUTHOR_LIST_COMPLETE, PUBMED_ID, CREATED_BY, CREATED_AT, MODIFIED_AT, MODIFIED_BY, TITLE_MARKUP,transaction_date, transaction_type) AS SELECT t.PAPER_ID, t.TITLE, t.PAGINATION, t.PAPERTYPE_ID, t.JOURNALISSUE_ID, t.BOOK_ID, t.AUTHOR_LIST_COMPLETE, t.PUBMED_ID, t.CREATED_BY, t.CREATED_AT, t.MODIFIED_AT, t.MODIFIED_BY, t.TITLE_MARKUP, coalesce(t.modified_at, t.created_at) as transaction_time, CASE WHEN t.modified_at IS NULL THEN 'I' ELSE 'U' END AS transaction_date FROM DB2INST1.PAPER t UNION SELECT a.PAPER_ID, a.TITLE, a.PAGINATION, a.PAPERTYPE_ID, a.JOURNALISSUE_ID, a.BOOK_ID, a.AUTHOR_LIST_COMPLETE, a.PUBMED_ID, a.CREATED_BY, a.CREATED_AT, a.MODIFIED_AT, a.MODIFIED_BY, a.TITLE_MARKUP,transaction_date, transaction_type FROM DB2INST1.AUDIT_PAPER a WHERE transaction_type = 'D'
