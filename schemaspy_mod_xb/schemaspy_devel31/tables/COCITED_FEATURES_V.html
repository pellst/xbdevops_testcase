View Definition:
CREATE VIEW db2inst1.cocited_features_v AS WITH feature_papers AS (SELECT a1.object_row_id as feature_id, a1.source_row_id AS paper_id FROM attribution a1 WHERE a1.object_name ='FEATURE' AND a1.source_name ='PAPER' AND a1.source_type = 'medium' ), cocited_features AS (SELECT f.feature_id as feature_id, f.name, a2.source_row_id as paper_id FROM feature f, attribution a2 WHERE f.feature_id = a2.object_row_id AND a2.object_name ='FEATURE' AND a2.source_name ='PAPER' AND a2.source_type ='medium' ), results as (select g.feature_id, c.feature_id as cocited_feature_id, c.name as cocited_name, count(*) as occurrences from feature_papers g, cocited_features c where g.paper_id = c.paper_id and g.feature_id != c.feature_id group by g.feature_id, c.feature_id, c.name) SELECT r.* FROM results r
