View Definition:
create view cocited_stage_genes_v AS WITH stage_papers AS (SELECT a1.object_row_id AS stage_id, a1.source_row_id AS paper_id FROM attribution a1 WHERE a1.object_name = 'STAGE' AND a1.source_name = 'PAPER' AND a1.source_type = 'medium'), cocited_features AS (SELECT f.feature_id AS feature_id, f.name, a2.source_row_id AS paper_id FROM feature f, attribution a2 WHERE f.feature_id = a2.object_row_id AND a2.object_name = 'FEATURE' AND a2.source_name = 'PAPER' AND a2.source_type = 'medium'), results AS (SELECT g.stage_id, c.feature_id AS cocited_feature_id, c.name AS cocited_name, COUNT(*) AS occurrences FROM stage_papers g, cocited_features c WHERE g.paper_id = c.paper_id GROUP BY g.stage_id, c.feature_id, c.name) SELECT r.* FROM results r
