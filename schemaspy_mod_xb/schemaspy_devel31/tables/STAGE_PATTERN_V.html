CREATE VIEW stage_pattern_v as ( select i.image_id, b.biomedia_id, b.experiment_id, bs.specimenstages_id, ss.specimen_id, ss.start_stage_id, ss.end_stage_id, s1.rank as start_stage_rank, s2.rank as end_stage_rank, est.expressionstagestested_id, est.expressiontest_id, et.feature_id as probe_id, pc.cdna_id, pc.cdna_name, pg.umbrella_gene_id, pg.umbrella_gene_symbol, pg.gene_id, pg.gene_symbol, coalesce(pg.umbrella_gene_id,pc.cdna_id) as expressed_feature_id, coalesce(pg.umbrella_gene_symbol,pc.cdna_name) as expressed_feature_name, e.experimentsource_id, CASE WHEN ( ( s1.name = 'unspecified stage' ) OR ( s2.name = 'unspecified stage' ) ) THEN s1.stage_number WHEN ( s1.name = s2.name ) THEN s1.name WHEN ( s1.name != s2.name ) THEN s1.name || ' to ' || s2.stage_number END AS stage_text from image i, biomedia b, biomedia_specimenstages bs, specimen_stages ss, stage s1, stage s2, expression_stages_tested est, expressiontest et left join probe_genes_v pg ON (et.feature_id=pg.probe_id) left join probe_cdnas_v pc ON (et.feature_id=pc.probe_id), experiment e where i.image_id=b.image_id AND bs.biomedia_id=b.biomedia_id AND ss.specimenstages_id=bs.specimenstages_id AND s1.stage_id=ss.start_stage_id AND s2.stage_id=ss.end_stage_id AND est.specimenstages_id=ss.specimenstages_id AND et.expressiontest_id=est.expressiontest_id AND e.experiment_id=b.experiment_id )
