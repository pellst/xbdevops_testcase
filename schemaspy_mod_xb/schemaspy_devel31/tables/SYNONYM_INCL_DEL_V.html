CREATE VIEW DB2INST1.SYNONYM_incl_del_v (SYNONYM_ID, NAME, TYPE_ID, SYNONYM_SGML, CREATED_BY, CREATED_AT, MODIFIED_BY, MODIFIED_AT,transaction_date, transaction_type) AS SELECT t.SYNONYM_ID, t.NAME, t.TYPE_ID, t.SYNONYM_SGML, t.CREATED_BY, t.CREATED_AT, t.MODIFIED_BY, t.MODIFIED_AT, coalesce(t.modified_at, t.created_at) as transaction_time, CASE WHEN t.modified_at IS NULL THEN 'I' ELSE 'U' END AS transaction_date FROM DB2INST1.SYNONYM t UNION SELECT a.SYNONYM_ID, a.NAME, a.TYPE_ID, a.SYNONYM_SGML, a.CREATED_BY, a.CREATED_AT, a.MODIFIED_BY, a.MODIFIED_AT,transaction_date, transaction_type FROM DB2INST1.AUDIT_SYNONYM a WHERE transaction_type = 'D'
