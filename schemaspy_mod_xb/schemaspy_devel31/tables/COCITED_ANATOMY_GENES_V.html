create view cocited_anatomy_genes_v AS WITH anatomy_item_papers AS (SELECT a1.object_row_id AS anatomy_item_id, a1.source_row_id AS paper_id FROM attribution a1 WHERE a1.object_name = 'ANATOMY_ITEM' AND a1.source_name = 'PAPER' AND a1.source_type = 'medium'), cocited_features AS (SELECT f.feature_id AS feature_id, f.name, a2.source_row_id AS paper_id FROM feature f, attribution a2 WHERE f.feature_id = a2.object_row_id AND a2.object_name = 'FEATURE' AND a2.source_name = 'PAPER' AND a2.source_type = 'medium'), results AS (SELECT g.anatomy_item_id, c.feature_id AS umbrella_gene_id, c.name AS umbrella_gene_symbol, g.paper_id FROM anatomy_item_papers g, cocited_features c WHERE g.paper_id = c.paper_id ) SELECT * FROM results
