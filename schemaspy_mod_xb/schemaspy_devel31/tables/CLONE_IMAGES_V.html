create view clone_images_v as ( select pc.cdna_id, pc.cdna_name, i.image_id, ss.start_stage_id, ss.end_stage_id, s1.rank as start_stage_rank, s2.rank as end_stage_rank, es.name as experimentsource, CASE WHEN ( ( s1.name = 'unspecified stage' ) OR ( s2.name = 'unspecified stage' ) ) THEN s1.stage_number WHEN ( s1.name = s2.name ) THEN s1.name WHEN ( s1.name != s2.name ) THEN s1.name || ' to ' || s2.stage_number END AS stage_text from image i, biomedia b, biomedia_specimenstages bs, specimen_stages ss, stage s1, stage s2, expression_stages_tested est, expressiontest et, probe_cdnas_v pc, experiment e, experimentsource es where et.feature_id=pc.probe_id AND est.expressiontest_id=et.expressiontest_id AND ss.specimenstages_id=est.specimenstages_id AND s1.stage_id=ss.start_stage_id AND s2.stage_id=ss.end_stage_id AND bs.specimenstages_id=ss.specimenstages_id AND b.biomedia_id=bs.biomedia_id AND i.image_id=b.image_id AND e.experiment_id=b.experiment_id AND es.experimentsource_id=e.experimentsource_id )
