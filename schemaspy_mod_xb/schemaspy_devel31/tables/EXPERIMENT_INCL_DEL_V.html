CREATE VIEW DB2INST1.EXPERIMENT_incl_del_v (EXPERIMENT_ID, EXPERIMENTSOURCE_ID, DESCRIPTION, CREATED_BY, CREATED_AT, MODIFIED_BY, MODIFIED_AT, TYPE_ID,transaction_date, transaction_type) AS SELECT t.EXPERIMENT_ID, t.EXPERIMENTSOURCE_ID, t.DESCRIPTION, t.CREATED_BY, t.CREATED_AT, t.MODIFIED_BY, t.MODIFIED_AT, t.TYPE_ID, coalesce(t.modified_at, t.created_at) as transaction_time, CASE WHEN t.modified_at IS NULL THEN 'I' ELSE 'U' END AS transaction_date FROM DB2INST1.EXPERIMENT t UNION SELECT a.EXPERIMENT_ID, a.EXPERIMENTSOURCE_ID, a.DESCRIPTION, a.CREATED_BY, a.CREATED_AT, a.MODIFIED_BY, a.MODIFIED_AT, a.TYPE_ID,transaction_date, transaction_type FROM DB2INST1.AUDIT_EXPERIMENT a WHERE transaction_type = 'D'
