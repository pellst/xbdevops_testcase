create or replace view db2inst1.has_frog_assay_info_v as with m1 as ( select paper_id,gse_id , count( gse_id ) over ( partition by paper_id ) gse_count , count(*) reccount from geo_data_profile_v where gse_id is not null group by paper_id, gse_id ), m2 as ( select * from m1 where gse_count > 1 ) select t2.* ,( select listagg( xptx.ref_name,';') WITHIN GROUP (ORDER BY xptx.curxpx_id) from db2inst1.cur_exp_experiment xptx where xptx.curxpx_id = t2.curxpx_id group by xptx.curxpx_id ) frog_info ,( select listagg( xptx.frog_info,';') WITHIN GROUP (ORDER BY xptx.curxpx_id) from db2inst1.has_xpx_frog_info_v xptx where xptx.curxpx_id = t2.curxpx_id group by xptx.curxpx_id ) frog_info_context ,( select listagg( xpa.ref_name,';') WITHIN GROUP (ORDER BY xpa.curxpa_id) from db2inst1.cur_exp_assay xpa where xpa.curxpa_id = t2.curxpa_id group by xpa.curxpa_id ) assay_info ,( select listagg( xpa.assay_info,';') WITHIN GROUP (ORDER BY xpa.curxpa_id) from db2inst1.has_xpa_assay_info_v xpa where xpa.curxpa_id = t2.curxpa_id group by xpa.curxpa_id ) assay_info_context from geo_data_profile_v t2 , m2 where m2.paper_id = t2.paper_id and m2.gse_id = t2.gse_id --and t2.tx_context is not null --order by t2.paper_id, t2.gse_id
